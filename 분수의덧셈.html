    class Solution {
    public int[] solution(int numer1, int denom1, int numer2, int denom2) {
        //분자
        int comnumer = (numer1 * denom2) + (numer2 * denom1);
        //분모 - 통분이라서 한번만함
        int comdenom = (denom1 * denom2);
        
        for(int i = comnumer-1; i > 0;i--){
         if(comnumer % i == 0 && comdenom % i == 0){
             comnumer /= i;
             comdenom /= i;
         }   
        }
      int []answer = {comnumer, comdenom};
        return answer;
    }
}

단계별 방법
분모의 최소공배수 구하기
두 분수의 분모의 최소공배수를 구합니다. 이를 통해 두 분수의 공통 분모를 만들 수 있습니다.

분자를 공통 분모에 맞게 변환
분모를 최소공배수로 통일하면, 각 분수의 분자도 그에 맞게 변환해야 합니다.

분자끼리 더하기
두 분수를 같은 분모로 변환했다면 이제 분자끼리 더할 수 있습니다.

기약분수로 만들기
마지막으로 유클리드 호제법을 사용해 분자와 분모의 최대공약수를 구해, 분자와 분모를 최대공약수로 나누어 기약분수로 만듭니다.

public class FractionAddition {

    // 유클리드 호제법을 이용한 최대공약수(GCD) 계산
    public static int gcd(int a, int b) {
        while (b != 0) {
            int temp = b;
            b = a % b;
            a = temp;
        }
        return a;
    }

    // 최소공배수(LCM) 계산
    public static int lcm(int a, int b) {
        return (a * b) / gcd(a, b);
    }

    public static int[] addFractions(int num1, int den1, int num2, int den2) {
        int commonDenominator = lcm(den1, den2);  // 공통 분모
        int newNum1 = num1 * (commonDenominator / den1);  // 첫 번째 분수의 분자 변환
        int newNum2 = num2 * (commonDenominator / den2);  // 두 번째 분수의 분자 변환

        int numeratorSum = newNum1 + newNum2;  // 분자끼리 더하기

        // 기약분수 만들기
        int gcdResult = gcd(numeratorSum, commonDenominator);
        return new int[]{numeratorSum / gcdResult, commonDenominator / gcdResult};  // [분자, 분모]
    }

    public static void main(String[] args) {
        int[] result = addFractions(1, 4, 1, 6);
        System.out.println("결과: " + result[0] + "/" + result[1]);
    }
}
